# std::initialize_rendering (a function)

  initializes the rendering of widgets

  Parameters:
    screen: the screen to use for rendering

  Result:
    continuation: a continuation that can be used to render widgets

  The function initializes the rendering of widgets by setting the
  template screen to use for rendering.

  Topic: Widgets

  See also: std::text_box, std::push_button, std::horizontal_group,
            std::vertical_group, std::table_group, std::yes_no_dialog,
            std::okay_dialog, std::okay_cancel_dialog

  Example:

    $screen create_screen(80 25)
    $continuation initialize_rendering(screen)
    $text_box text_box(continuation "This is a text box")
    $push_button push_button(continuation "Ok" = OKAY)
    $horizontal_group horizontal_group(text_box push_button)
    $vertical_group vertical_group(text_box push_button)
    $table_group table_group(list(text_box push_button) list(push_button text_box))
    $yes_no_dialog yes_no_dialog(continuation "Question" "Is this a question?")
    $okay_dialog okay_dialog(continuation "Message" "This is a message.")
    $okay_cancel_dialog okay_cancel_dialog(continuation "Message" "This is a message.")
